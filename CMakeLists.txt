project(Cyanite)

set(CYANITE Cyanite)
set(NAME "Cyanite Game Engine")
set(Version 0.0.1a)

    ${CYANITE}
    PROPERTIES
    MACOSX_BUNDLE_INFO_PLIST
    resources/plist.in
)
